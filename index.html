//variáveis da bolinha let xBolinha = 300; let yBolinha = 200; let diametro = 30; let raio = diametro/2; let velocidadeXBolinha = 7; let velocidadeYBolinha = 8; //variáveis da raquete larguraRaquete = 10; alturaRaquete = 100; xRaquete = 10; yRaquete = 150; xRaqueteOponente = 580; yRaqueteOponente = 150; function setup() { createCanvas(600, 400); // background (0); } function draw() { background (0); bolinha(); movimentaBolinha(); colisaoBolinhaBorda(); mostraRaquete(); mostraRaqueteOponente(); } function bolinha(){ circle(xBolinha,yBolinha,diametro); } function movimentaBolinha(){ xBolinha += velocidadeXBolinha; yBolinha += velocidadeYBolinha; } function colisaoBolinhaBorda(){ if(xBolinha + raio > width || xBolinha - raio < 0){ velocidadeXBolinha *= -1; } if(yBolinha + raio > height || yBolinha - raio < 0){ velocidadeYBolinha *= -1; } } function mostraRaquete(){ rect(xRaquete,yRaquete,larguraRaquete,alturaRaquete) } function mostraRaqueteOponente(){ rect(xRaqueteOponente, yRaqueteOponente,larguraRaquete,alturaRaquete) }
